880 ConnectionT<T> * makeConnection(InputPort<T> * a, OutputPort<T> * b, ConnectionPolicy p) {
881  switch(p.lock_policy_) {
882  case ConnectionPolicy::LOCKED:
883  switch (p.data_policy_) {
884  case ConnectionPolicy::DATA: return new ConnectionDataL<T>(a,b,p);
885  case ConnectionPolicy::BUFFER: return new ConnectionBufferL<T>(a,b,p);
886  case ConnectionPolicy::CIRCULAR: return new ConnectionBufferL<T>(a,b,p);
887  }
888  break;
889  case ConnectionPolicy::UNSYNC:
890  switch (p.data_policy_) {
891  case ConnectionPolicy::DATA: return new ConnectionDataU<T>(a,b,p);
892  case ConnectionPolicy::BUFFER: return new ConnectionBufferU<T>(a,b,p);
893  case ConnectionPolicy::CIRCULAR: return new ConnectionBufferU<T>(a,b,p);
894  }
895  break;
896  case ConnectionPolicy::LOCK_FREE:
897  switch (p.data_policy_) {
898  case ConnectionPolicy::DATA:
899  p.buffer_size_ = 1;
900  p.data_policy_ = ConnectionPolicy::CIRCULAR;
901  return new ConnectionBufferL<T>(a,b,p);
902  //return new ConnectionBufferLF<T>(a,b,p);
903  case ConnectionPolicy::BUFFER: return new ConnectionBufferL<T>(a,b,p);
904  case ConnectionPolicy::CIRCULAR: return new ConnectionBufferL<T>(a,b,p);
905  //case ConnectionPolicy::BUFFER: return new ConnectionBufferLF<T>(a,b,p);
906  //case ConnectionPolicy::CIRCULAR: return new ConnectionBufferLF<T>(a,b,p);
907  }
908  break;
909  }
910  return 0;
911 }
